\frame{
\frametitle{Biochemický priestor}

\textbf{Biochemický priestor} (Biochemical Space -- BCS) je doména biologických znalostí s formálnymi prvkami, ktorá poskytuje 

\begin{itemize}
\item \textbf{popis},
\item anotáciu,
\item zdieľanie
\end{itemize}

špecificky zameraných biologických modelov.

}

% -------------------------------------------------------------

\frame{
\frametitle{Ciele práce}
  
\begin{center}
\begin{enumerate}
	\item formálna definícia jazyka Biochemického priestoru,
	\begin{itemize}
		\item kompaktný zápis chemických reakcií tzv. \emph{rule-based} prístupom,
		\item abstrakcia od detailov chemických väzieb,
		\item hierarchické usporiadanie individuálych látok,
		\item spustiteľná sémantika,
	\end{itemize}
	\item návrh analyzačných techník pre tento jazyk.
\end{enumerate}

\noindent\makebox[\linewidth]{\rule{1.1\textwidth}{0.4pt}}

Nad rámec zadania

\begin{enumerate}
	\item prototypová implementácia na editovanie a analyzovanie modelov zapísaných v tomto jazyku.
\end{enumerate}
\end{center}

}

\frame{
\frametitle{Definované analýzy}
\begin{itemize}
\item Odstránenie redundantných pravidiel

\begin{itemize}
	\item zvýšenie jednoduchosti a prehľadnosti modelu
\end{itemize}

\item Redukcia kontextu

\begin{itemize}
	\item odstránenie stavov z pravidiel -- pravidlá vytvárania a zániku komplexov
\end{itemize}

\item Statická analýza \textbf{ne}dosažiteľnosti

\begin{itemize}
	\item overenie \textbf{ne}dosažiteľnosti na základe vlastnosti definovanej na pravidlách
\end{itemize}

\end{itemize}
}

\frame{
\frametitle{Nástroj BCSgen}

Prototypová implementácia na editovanie a analyzovanie modelov zapísaných v jazyku Biochemického priestoru.

Funkcie

\begin{itemize}
\item interaktívny editor modelov,
\item generovanie prechodového systému,
\item generovanie odvodených reakcií z daných pravidiel,
\item overovanie dosažiteľnosti skupiny zvolených látok,
\item vizualizácia prechodových systémov (vrátane dosažiteľnosti),
\item simulovanie modelov s definovanou kinetikou.
\item ukladanie a načítanie modelov/prechodových systémov/vizualizácií.
\end{itemize}

}
